library ieee;
use ieee.std_logic_1164.all;

entity lock_input is
    port
    (
        clk             : in  std_logic;
        reset_n         : in  std_logic;

        KEY             : in  std_logic_vector(3 downto 0);  -- active LOW buttons
        SW0             : in  std_logic; -- active HIGH
        SW1             : in  std_logic;

        key_press       : out std_logic_vector(3 downto 0);  -- pulse when button pressed
        enable_entry    : out std_logic;                     -- SW0 registered
        clear_entry     : out std_logic                      -- pulse on SW1 rising edge
    );
end entity lock_input;
 
architecture rtl of lock_input is

    signal key_sync0  : std_logic_vector(3 downto 0);
    signal key_sync1  : std_logic_vector(3 downto 0);
    signal key_prev   : std_logic_vector(3 downto 0);

    signal sw1_prev   : std_logic;

    signal key_pulse_i : std_logic_vector(3 downto 0);
    signal sw1_pulse_i : std_logic;

begin

process(clk, reset_n)
begin

    if reset_n = '0' then

        key_sync0 <= (others => '1');
        key_sync1 <= (others => '1');
        key_prev  <= (others => '0');

        sw1_prev  <= '0';

        key_pulse_i <= (others => '0');
        sw1_pulse_i <= '0';

    elsif rising_edge(clk) then

        -- convert active LOW buttons to active HIGH
        key_sync0 <= not KEY;
        key_sync1 <= key_sync0;

        -- detect button presses
        key_pulse_i(0) <= key_sync1(0) and not key_prev(0);
        key_pulse_i(1) <= key_sync1(1) and not key_prev(1);
        key_pulse_i(2) <= key_sync1(2) and not key_prev(2);
        key_pulse_i(3) <= key_sync1(3) and not key_prev(3);

        key_prev <= key_sync1;

        -- detect SW1 rising edge
        sw1_pulse_i <= SW1 and not sw1_prev;
        sw1_prev    <= SW1;

    end if;

end process;

key_press    <= key_pulse_i;
enable_entry <= SW0;
clear_entry  <= sw1_pulse_i;

end architecture rtl;
